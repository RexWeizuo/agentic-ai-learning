<script setup>
// 从 Vue 库中导入 ref 函数，ref 用于创建响应式数据
import { ref } from 'vue'

// 使用 ref 创建一个响应式变量 value，初始值为空字符串
// 这个变量将用于绑定输入框的值
const value = ref('')

// 使用 ref 创建一个响应式数组 list，初始包含三个待办事项对象
// 每个对象有两个属性：value 表示事项内容，isCompleted 表示事项是否完成
const list = ref([
  { value: '吃饭', isCompleted: true },
  { value: '睡觉', isCompleted: false },
  { value: '打豆豆', isCompleted: false },
])

// 定义一个函数 add，用于添加新的待办事项
function add() {
  // 向 list 数组中添加一个新的待办事项对象
  // 新对象的 value 属性为输入框的值，isCompleted 为 false
  list.value.push({
    value: value.value,
    isCompleted: false,
  })

  // 添加完成后，将输入框的值清空
  value.value = ''
}

// 定义一个函数 del，用于删除指定索引的待办事项
function del(index) {
  // 使用 splice 方法从 list 数组中移除指定索引的元素
  list.value.splice(index, 1)
}
</script>
<template>
  <!-- 定义一个根 div，类名为 todo-app，用于包裹整个待办事项应用 -->
  <div class="todo-app">
    <!-- 显示应用标题 -->
    <div class="title">Todo App</div>

    <!-- 定义一个表单区域，用于输入新的待办事项 -->
    <div class="todo-form">
      <!-- 输入框，使用 v-model 指令将输入框的值绑定到 value 变量 -->
      <!-- 输入框类型为文本，类名为 todo-input，占位符提示用户输入待办事项 -->
      <input
        v-model="value"
        type="text"
        class="todo-input"
        placeholder="Add a todo"
      />
      <!-- 点击该 div 会触发 add 函数，添加新的待办事项 -->
      <!-- 类名为 todo-button，显示文本为 Add Todo -->
      <div @click="add" class="todo-button">Add Todo</div>
    </div>

    <!-- 使用 v-for 指令遍历 list 数组，渲染每个待办事项 -->
    <!-- 根据事项的完成状态动态绑定类名，完成则添加 completed 类，否则添加 item 类 -->
    <div
      v-for="(item, index) in list"
      :class="[item.isCompleted ? 'completed' : 'item']"
    >
      <div>
        <!-- 复选框，使用 v-model 指令将复选框的选中状态绑定到 item.isCompleted -->
        <input v-model="item.isCompleted" type="checkbox" />
        <!-- 显示待办事项的内容 -->
        <span class="name">{{ item.value }}</span>
      </div>

      <!-- 点击该 div 会触发 del 函数，删除当前索引的待办事项 -->
      <!-- 类名为 del，显示文本为 del -->
      <div @click="del(index)" class="del">del</div>
    </div>
  </div>
</template>
