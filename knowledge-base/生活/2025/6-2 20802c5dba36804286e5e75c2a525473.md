        response = client.chat.completions.create(
            model="deepseek-chat",
            messages=context,
            temperature=0.3,
            max_tokens=2000
        )
        return response.choices[0].message.content.strip()
    except Exception as e:
        print(f"API调用失败: {str(e)}")
